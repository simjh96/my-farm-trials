SELECT mem.nickname, password,
		enabled, mem.kakaopk, kakaothumb, kakaoname, email, zipcode, address,
		phone,
		regdate,
		auth.auth FROM MYNGUSER mem LEFT OUTER JOIN MYNGUSER_AUTH
		auth on
		mem.kakaopk = auth.kakaopk WHERE mem.nickname = 'test1';

CREATE TABLE MYNGUSER (
    NO NUMBER(4) NOT NULL,
    NICKNAME VARCHAR2(100),
    PASSWORD VARCHAR2(500),
    ENABLED CHAR(1) DEFAULT '1',
    KAKAOPK VARCHAR2(500) NOT NULL CONSTRAINT MYNGUSER_PK PRIMARY KEY,
    KAKAOTHUMB VARCHAR2(500) NOT NULL,
    KAKAONAME VARCHAR2(500) NOT NULL,
    EMAIL VARCHAR2(500),
    ZIPCODE NUMBER(10),
    ADDRESS VARCHAR2(500),
    PHONE VARCHAR2(500),
    REGDATE DATE
);

CREATE TABLE MYNGUSER_AUTH (
    NO NUMBER(4) NOT NULL,
    KAKAOPK VARCHAR2(500) NOT NULL CONSTRAINT MYNGUSER_AUTH_PK PRIMARY KEY,
    AUTH VARCHAR2(50) NOT NULL,
    CONSTRAINT FK_MYNGUSER_AUTH FOREIGN KEY(KAKAOPK) REFERENCES MYNGUSER(KAKAOPK)
);
